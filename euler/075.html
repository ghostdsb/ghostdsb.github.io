import mathtools,math

lim = 15*10**5
triples = []
for s in range(3,math.floor(math.sqrt(lim)),2):
	for t in range(s-2,0,-2):
		if math.gcd(s,t)==1:
			a = s*t
			b = (s*s-t*t)//2
			c = (s*s+t*t)//2
			triples.append([a,b,c])

vals = [0]*(lim+1)

for triple in triples:
	summ = sum(triple)
	for i in range(summ,len(vals),summ):
		vals[i] += 1
		
ans = 0
for i in vals:
	if i==1:
		ans += 1
		
print(ans)
