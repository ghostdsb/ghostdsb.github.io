def digits(num):
	n = num
	dig = 0
	while n > 0:
		n = n//10
		dig += 1
	return dig

def count_char(word):
	w = 'qwertyuioplkjhgfdsazxcvbnm'
	chars = 0
	for l in word:
		if l in w:
			chars += 1
	return chars

def two_digit(i):
	if i not in words:
		tens = i//10
		units = i%10
		ans = words[tens*10]+words[units]
	if i in words:
		ans = words[i]
	return ans

tot_chars = 0	

for i in range(1,1001):	
	if digits(i)==1:
		ans = words[i]
	elif digits(i)==2:
		ans = two_digit(i)
	elif digits(i)==3 and i not in words:
		t_digit = i%100
		hundreds = i//100
		if t_digit != 0:
			ans = words[hundreds]+words[100]+" and "+two_digit(t_digit)

	tot_chars += count_char(ans)
