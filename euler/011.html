def prod(ox, oy, dx, dy, n):
	result = 1
	for i in range(n):
		result *= numbers[oy + i*dy][ox + i*dx]
	return result

numbers = [[int(x) for x in line.split()] for line in given_num.split("\n")]
adjacent = 4
ans = -1
xlen = len(numbers[0])
ylen = len(numbers)
for y in range(ylen):
	for x in range(xlen):
		if x + adjacent <= xlen:
			ans = max(prod(x, y,  1, 0, adjacent), ans)
		if y + adjacent <= ylen:
			ans = max(prod(x, y,  0, 1, adjacent), ans)
		if x + adjacent <= xlen and y + adjacent <= ylen:
			ans = max(prod(x, y,  1, 1, adjacent), ans)
		if x - adjacent >= -1    and y + adjacent <= ylen:
			ans = max(prod(x, y, -1, 1, adjacent), ans)
print(ans)
