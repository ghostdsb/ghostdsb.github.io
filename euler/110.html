# 1/(n+a) + 1/(n+b) = 1/n
# n^2 = ab
import math,mathtools
i = 1
while True:
	n_sols = len(mathtools.divisorlist(i*i))
	if n_sols % 2 == 0:
		n_sols /= 2
	else:
		n_sols = (n_sols+1)/2
	if n_sols > 1000:
		ans = i
		break
	i+=1
print(ans)
